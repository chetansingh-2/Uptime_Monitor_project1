        document.addEventListener('DOMContentLoaded', function() {

                // Read the data from the data-* attributes
                const chartData = JSON.parse(el.dataset.sparkline || '[]');
                // console.log('Sparkline Data:', chartData);
                const chartColor = el.dataset.color || '#94a3b8';
                
                const options = {
                    chart: {
                        type: 'area',
                        height: 80,
                        sparkline: { enabled: true },
                        background: 'transparent',
                        animations: { enabled: true, easing: 'easeinout', speed: 400 }
                    },
                    stroke: { curve: 'smooth', width: 2 },
                    fill: {
                        type: 'gradient',
                        gradient: {
                            shade: 'dark',
                            type: 'vertical',
                            opacityFrom: 0.5,
                            opacityTo: 0.1
                        }
                    },
                    series: [{ name: 'Response Time', data: chartData }],
                    colors: [chartColor],
                    tooltip: {
                        enabled: true,
                        theme: 'dark',
                        y: {
                            formatter: function(val) { return val + 's'; }
                        }
                    },
                    grid: { show: false },
                    xaxis: { labels: { show: false } },
                    yaxis: { labels: { show: false } }
                };

                const chart = new ApexCharts(el, options);
                chart.render();
            });
        });
